-- Availability Snapshot Query:

SELECT difficulty_band,
       COUNT(*) FILTER (WHERE pyq_frequency_score=1.5) AS pyq15,
       COUNT(*) FILTER (WHERE pyq_frequency_score=1.0) AS pyq10,
       COUNT(*) AS total
FROM questions
GROUP BY 1 ORDER BY 1;
