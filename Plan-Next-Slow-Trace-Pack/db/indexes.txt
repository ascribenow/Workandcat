-- Database Indexes for Hot Tables


-- Table: attempt_events
CREATE INDEX idx_attempt_events_sess_seq ON public.attempt_events USING btree (sess_seq_at_serve)
CREATE INDEX idx_attempt_events_user_sess ON public.attempt_events USING btree (user_id, sess_seq_at_serve)
CREATE INDEX idx_attempt_events_user_session ON public.attempt_events USING btree (user_id, session_id)
CREATE UNIQUE INDEX attempt_events_pkey ON public.attempt_events USING btree (id)

-- Table: questions
CREATE INDEX idx_questions_concept_extraction_status ON public.questions USING btree (concept_extraction_status)
CREATE INDEX idx_questions_pair ON public.questions USING btree (subcategory, type_of_question)
CREATE INDEX idx_questions_quality_verified ON public.questions USING btree (quality_verified)
CREATE INDEX idx_questions_solution_method ON public.questions USING btree (solution_method)
CREATE UNIQUE INDEX questions_pkey ON public.questions USING btree (id)

-- Table: session_pack_plan
CREATE INDEX idx_pack_plan_user_sess ON public.session_pack_plan USING btree (user_id, session_id)
CREATE INDEX idx_session_pack_plan_created_at ON public.session_pack_plan USING btree (created_at)
CREATE INDEX idx_session_pack_plan_status ON public.session_pack_plan USING btree (status)
CREATE INDEX idx_session_pack_plan_user_id ON public.session_pack_plan USING btree (user_id)
CREATE UNIQUE INDEX session_pack_plan_pkey ON public.session_pack_plan USING btree (user_id, session_id)
CREATE UNIQUE INDEX uq_pack_plan_planned ON public.session_pack_plan USING btree (user_id, session_id, status) WHERE ((status)::text = 'planned'::text)
